return nil;
// 将 span 插到 empty 链表。
runtime·MSpanList_InsertBack(&c->empty, s);
havespan:
// 设置待返回 span 的相关属性。
cap = (s->npages << PageShift) / s->elemsize;
n = cap - s->ref;
// 标记被 cache 使󲙢。
s->incache = true;
return s;
}
