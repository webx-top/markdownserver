}
exit <- true
}()
data <- 4 // 如果缓冲区已满，阻塞。
data <- 5
close(data)
<-exit
}
缓冲区是内部属性，并󵻸类型构成要素。
var a, b chan int = make(chan int), make(chan int, 3)
除󲙢 range 外，还可󲙢 ok-idiom 模式判断 channel 是否关闭。
for {
if d, ok := <-data; ok {
fmt.Println(d)
} else {
break
}
}

内置函数 len 返回未被读取的缓冲元素数量，cap 返回缓冲区。
d1 := make(chan int)
d2 := make(chan int, 3)
d2 <- 1
fmt.Println(len(d1), cap(d1)) // 0 0
fmt.Println(len(d2), cap(d2)) // 1 3
