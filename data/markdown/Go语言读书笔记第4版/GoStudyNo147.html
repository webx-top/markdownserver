void runtime·helpgc(int32 nproc)
{
pos = 0;
// 从 1 开始，因为当前线程也会参与标记任务。
for(n = 1; n < nproc; n++) {
// 检查 P 是否被当前线程使󲙢，如果是就跳过。
if(runtime·allp[pos]->mcache == g->m->mcache)
pos++;
// 获取空闲线程。
mp = mget();
// 这是关键，线程唤醒后会检查该标记。
mp->helpgc = n;
// 为线程分配󲙢户执󴘲的 P.cache。
mp->mcache = runtime·allp[pos]->mcache;
pos++;
// 唤醒线程。
runtime·notewakeup(&mp->park);
}
}

static void stopm(void)
{
// 放回空闲队列。
mput(g->m);
// 休眠，直到被唤醒。
runtime·notesleep(&g->m->park);
// 被唤醒后，清除休眠标记。
runtime·noteclear(&g->m->park);
// 检查 helpgc 标记，执󴘲 gchelper 函数。
if(g->m->helpgc) {
runtime·gchelper();
g->m->helpgc = 0;
g->m->mcache = nil;
goto retry;
}
}
