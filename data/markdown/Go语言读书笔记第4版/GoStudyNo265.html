还可󲙢 net/http/pprof 实时查看 runtime profiling 信息。
package main
import (
_ "net/http/pprof"
"net/http"
"time"
)
func main() {
go http.ListenAndServe("localhost:6060", nil)
for {
time.Sleep(time.Second)
}
}
在浏览器中查看 http://localhost:6060/debug/pprof/。
