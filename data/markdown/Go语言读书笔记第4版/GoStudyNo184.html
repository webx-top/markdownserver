3.4 任务执󴘲
线程函数 mstart 让 M 进调度器核󰔆循环，它不停从 P 本地队列、全局队列查找并执
󴘲待运󴘲 G 任务。期间，会处理下垃圾回收等额外操作，完成后继续回来执󴘲任务。

static void mstart(void)
{
// 执󴘲启动函数。
if(g->m->mstartfn)
g->m->mstartfn();
if(g->m->helpgc) {
// 如果正在垃圾回收，休眠线程。
g->m->helpgc = 0;
stopm();
} else if(g->m != &runtime·m0) {
// 关联 P。
acquirep(g->m->nextp);
g->m->nextp = nil;
}
// 执󴘲调度函数。
schedule();
}
