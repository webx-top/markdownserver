gp := sg.g
unlock(&c.lock)
goready(gp)
}
selected = true
received = true
return
}

func closechan(c *hchan) {
// 不要 close nil channel。
if c == nil {
panic("close of nil channel")
}
// 不要多次 close channal。
if c.closed != 0 {
panic("close of closed channel")
}
// 关闭标志。
c.closed = 1
// 唤醒所有排队的接收者。
for {
sg := c.recvq.dequeue()
if sg == nil {
break
}
gp := sg.g
sg.elem = nil
gp.param = nil // 如果是同步󰸵式，表明是 closechan 唤醒。
goready(gp)
}
// 唤醒所有排队的发送者。
for {
sg := c.sendq.dequeue()
if sg == nil {
break
}
