// 从链表提取个 Defer。
d := gp._defer
if d == nil {
return
}
// 调󲙢 deferproc 后，会 pop 掉 siz、fn，那么 arg0 就是 argp。
// 如果地址不等，显然就属于󰺂效调󲙢。
argp := uintptr(unsafe.Pointer(&arg0))
if d.argp != argp {
return
}
mp := acquirem()
// 复制参数。
// 很󰺂语，这出了个 deferArgs，和前󵼃的 d+1 个意思。
// 这真的是同个写的代码？
memmove(unsafe.Pointer(argp), deferArgs(d), uintptr(d.siz))
fn := d.fn
d.fn = nil
gp._defer = d.link
freedefer(d)
releasem(mp)
// 执󴘲 defer 函数。
jmpdefer(fn, argp)
}
