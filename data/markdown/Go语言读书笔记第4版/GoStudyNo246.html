runtime·finq = block;
}
// 添加到 FinBlock 队列。
f = &runtime·finq->fin[runtime·finq->cnt];
runtime·finq->cnt++;
f->fn = fn;
f->nret = nret;
f->fint = fint;
f->ot = ot;
f->arg = p;
// 有了新任务，设置唤醒标记。
runtime·fingwake = true;
}
