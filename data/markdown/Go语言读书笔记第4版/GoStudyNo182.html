// 临时保存待󲙢 P。
mp->nextp = p;
// 唤醒。
runtime·notewakeup(&mp->park);
}
static M* mget(void)
{
// 从空闲列表获取 M。
if((mp = runtime·sched.midle) != nil){
runtime·sched.midle = mp->schedlink;
runtime·sched.nmidle--;
}
return mp;
}
