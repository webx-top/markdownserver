bitp += step;
if(step == 0) {
// 8-byte objects.
bitp++;
shift = gcBits;
}
// 遍历该 span 所有 object。
for(; n > 0; n--, p += size) {
// 获取标记位。
bitp -= step;
if(step == 0) {
if(shift != 0)
bitp--;
shift = gcBits - shift;
}
xbits = *bitp;
bits = (xbits>>shift) & bitMask;
// 如果 object 对象标记为可达 (Marked)，则跳过。
// 包括 freelist 󵛜的未使󲙢对象。
if((bits&bitMarked) != 0) {
*bitp &= ~(bitMarked<<shift);
continue;
}
// 重置标记位。
*bitp = (xbits & ~((bitMarked|(BitsMask<<2))<<shift)) |
((uintptr)BitsDead<<(shift+2));
if(cl == 0) { // 对象。
// 清除全部标记位。
runtime·unmarkspan(p, s->npages<<PageShift);
// 重置代龄。
runtime·atomicstore(&s->sweepgen, sweepgen);
sweepgenset = true;
if(runtime·debug.efence) {
// ...
} else
// 将对象所使󲙢的 span 归还给 heap。
runtime·MHeap_Free(&runtime·mheap, s, 1);

runtime·xadd64(&mstats.next_gc,
-(uint64)(size * (runtime·gcpercent + 100)/100));
